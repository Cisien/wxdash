Description=wxdash Weather Kiosk

**4. Modify `src/CMakeLists.txt`** — add install rules after the existing target_link_libraries line:

```cmake
# --- Install rules ---
include(GNUInstallDirs)

# Install the wxdash executable
install(TARGETS wxdash
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    COMPONENT kiosk
)

# Deploy user-built QML module (wxdash URI -> Main.qml, ArcGauge.qml, etc.)
# NO_UNSUPPORTED_PLATFORM_ERROR: prevents fatal CMake error on Linux embedded
# DEPLOY_USER_QML_MODULES_ON_UNSUPPORTED_PLATFORM: ensures wxdash QML module lands next to binary
qt_generate_deploy_qml_app_script(
    TARGET wxdash
    OUTPUT_SCRIPT deploy_script
    NO_UNSUPPORTED_PLATFORM_ERROR
    DEPLOY_USER_QML_MODULES_ON_UNSUPPORTED_PLATFORM
    NO_TRANSLATIONS
)
install(SCRIPT ${deploy_script}
    COMPONENT kiosk
)
```

**5. Modify `CMakeLists.txt` (top-level)** — add install rules AFTER `add_subdirectory(tests)`:

```cmake
# --- Install rules (kiosk deployment) ---
include(GNUInstallDirs)

# Configure systemd service file (substitutes @CMAKE_INSTALL_PREFIX@)
configure_file(
    ${CMAKE_SOURCE_DIR}/assets/wxdash.service.in
    ${CMAKE_BINARY_DIR}/wxdash.service
    @ONLY
)

# Configure desktop entry (substitutes @CMAKE_INSTALL_PREFIX@)
configure_file(
    ${CMAKE_SOURCE_DIR}/assets/wxdash-kiosk.desktop.in
    ${CMAKE_BINARY_DIR}/wxdash-kiosk.desktop
    @ONLY
)

# Install systemd unit file
install(FILES ${CMAKE_BINARY_DIR}/wxdash.service
    DESTINATION lib/systemd/system
    COMPONENT kiosk
)

# Install desktop entry files
install(FILES
    assets/wxdash.desktop
    ${CMAKE_BINARY_DIR}/wxdash-kiosk.desktop
    DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/applications
    COMPONENT kiosk
)

# Install SVG icon to hicolor theme
install(FILES assets/wxdash.svg
    DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/icons/hicolor/scalable/apps
    COMPONENT kiosk
)

# Install EGLFS KMS config for Pi 4/5
install(FILES assets/eglfs.json
    DESTINATION ${CMAKE_INSTALL_DATAROOTDIR}/wxdash
    COMPONENT kiosk
)

# Convenience install-kiosk target: installs + enables systemd service
# Requires: sudo cmake --build build --target install-kiosk
add_custom_target(install-kiosk
    COMMAND ${CMAKE_COMMAND} --install ${CMAKE_BINARY_DIR} --component kiosk
    COMMAND systemctl daemon-reload
    COMMAND systemctl enable wxdash.service
    COMMENT "Installing wxdash kiosk and enabling systemd service (requires sudo)"
    VERBATIM
)
```

**Important notes for executor:**
- Do NOT use `qt_generate_deploy_app_script` without `NO_UNSUPPORTED_PLATFORM_ERROR` — it causes a fatal CMake error on Linux.
- The existing `assets/wxdash-kiosk.desktop` file should NOT be deleted — it serves as the current dev convenience file. The `.in` template is a separate file for installed deployments.
- `COMPONENT kiosk` on all install rules allows selective installation via `cmake --install build --component kiosk`.
- `configure_file(@ONLY)` only substitutes `@VAR@` patterns, not `${VAR}` — safe for service file content.
